2. Add to GitHub Actions
Create .github/workflows/quantum-shield.yml
name: Quantum Shield Scan
on: [push, pull_request]

jobs:
  security-scan:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Run Quantum Shield
        run: |
          zip -r code.zip . -x "*.git*"
          
          curl -X POST "https://quantum-shield.onrender.com/scan" \
            -H "x-api-key: ${{ secrets.QS_API_KEY }}" \
            -F "file=@code.zip"
